/**
 * Function to sort the date by order
 * Function compare
 * @param {string} a - date a;
 * @param {string} b - date b
 * @returns comparison
 */

function compare(a, b) {
    const dateA = a.d;
    const dateB = b.d;

    let comparison = 0;
    if (dateA > dateB) {
        comparison = 1;
    } else if (dateA < dateB) {
        comparison = -1;
    }
    return comparison;
}
